  public static function testLoadFromFileComparison ( $filename ) {
  
    $string = file_get_contents ( $filename );
  
    $filePattern = MapFilter_Pattern::fromFile ( $filename );
    $stringPattern = MapFilter_Pattern::load ( $string );
    
    self::assertEquals (
        $filePattern,
        $stringPattern
