Parse .ini file example 

<!--
  parse_ini_file () example from:
  http://www.php.net/manual/en/function.parse-ini-file.php
  
  There is an anitialization file parse_ini_file.ini with following content:
-->
<pattern>
  <all>
    <node_attr attr="first_section">
      <all assert="no_any_of_that">
        <attr assert="no_one">one</attr>
        <attr assert="no_five">five</attr>
        <attr assert="no_animal">animal</attr>
      </all>
    </node_attr>
    <node_attr attr="second_section">
      <all>
        <attr>path</attr>
        <attr>URL</attr>
      </all>
    </node_attr>
    <node_attr attr="third_section">
      <attr iterator="auto">phpversion</attr>
    </node_attr>
  </all>
</pattern>

; This is a sample configuration file
; Comments start with ';', as in php.ini

[first_section]
one = 1
five = 5
animal = Dodo bird

[second_section]
path = "/usr/local/bin"
URL = "http://www.example.com/~username"

[third_section]
phpversion[] = "5.0"
phpversion[] = "5.1"
phpversion[] = "5.2"
phpversion[] = "5.3"

  public static function testParse () {
  
    $content = parse_ini_file (
        PHP_MAPFILTER_TEST_DIR . MapFilter_Test_Sources::PARSEINIFILE_INI,
        self::EXPAND_SECTIONS
    );

    $pattern = MapFilter_TreePattern::fromFile (
        PHP_MAPFILTER_TEST_DIR . MapFilter_Test_Sources::PARSEINIFILE_XML
    );
    
    $filter = new MapFilter (
        $pattern,
        $content
    );
    
    $result = $filter->fetchResult ();

    self::assertEquals (
        $content,
        $result->getResults ()
    );
  }
